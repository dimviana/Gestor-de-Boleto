# Stage 1: Builder - Installs all dependencies, including dev, and builds the app
FROM node:lts-alpine AS builder

# Install necessary build tools and Python with venv support
RUN apk add --no-cache python3 py3-pip py3-venv make g++

WORKDIR /usr/src/app

# Set up Python virtual environment
RUN python3 -m venv /opt/venv
# Make sure pip is up-to-date and install PyMuPDF inside the venv
RUN /opt/venv/bin/pip install --upgrade pip && \
    /opt/venv/bin/pip install PyMuPDF

# Copy package files
COPY package.json ./

# Install all Node.js dependencies
RUN npm install

# Copy the rest of the application source code
COPY . .

# Build the TypeScript project
RUN npm run build

# Stage 2: Production - A lean image with only production code and dependencies
FROM node:lts-alpine

WORKDIR /usr/src/app

# Install pm2 globally and Python runtime
RUN npm install -g pm2 && apk add --no-cache python3

# Copy production dependencies from the builder stage
COPY --from=builder /usr/src/app/package.json ./
COPY --from=builder /usr/src/app/node_modules ./node_modules

# Copy the compiled code from the builder stage
COPY --from=builder /usr/src/app/dist ./dist

# Copy the Python script and virtual environment
COPY --from=builder /usr/src/app/api/services/parser.txt ./api/services/parser.txt
COPY --from=builder /opt/venv /opt/venv

# Copy database schema if needed for volume mounts, etc.
COPY --from=builder /usr/src/app/database/schema.txt ./database/schema.txt

# Expose the port the app runs on
EXPOSE 3001

# Start the application using pm2-runtime
# This is the recommended way to run PM2 in a Docker container
CMD [ "pm2-runtime", "dist/api/index.js" ]
